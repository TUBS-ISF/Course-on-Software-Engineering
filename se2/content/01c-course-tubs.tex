\begin{frame}
	\frametitle{Grade Improvement via Klausur+}

	\begin{fancycolumns}[widths={55}]
		\begin{definition}{The Exercises}
			\begin{itemize}
				\item "Klausur+" can improve your exam grade (requires passing)
				\item Can be improved by up-to two grade steps (e.g., from 2.3 to 1.7)
				\item First step: Participation in exercises
				\begin{itemize}
					\item Work in small groups
					\item Present solutions in exercises
				\end{itemize}
				\item Second step: Submit video on lecture topic
				\item Acquire any combination of first and second step to acquire one or two steps
			\end{itemize}
		\end{definition}
		\nextcolumn
		\begin{definition}{Video Guidelines}
			\begin{itemize}
				\item 5--10 minutes
				\item Topics: choose from list or suggest your own
				\item Submit for exercise corresponding to topic
				\item Earliest videos for topic discussed in start of May
				\item List of topics will be announced via StudIP
				\item Distribution via first come, first serve
			\end{itemize}
		\end{definition}
	\end{fancycolumns}

\end{frame}
