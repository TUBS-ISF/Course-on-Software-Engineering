\begin{frame}{\insertsubsection}
	\begin{fancycolumns}
		\begin{definition}{Composite \mysource{\gof}}
			\setlength\tabcolsep{1mm}
			\begin{tabularx}{\textwidth}{rX}
				intent     & \mycite{Compose objects into tree structures to represent part-whole hierarchies. Composite lets clients treat individual objects and compositions of objects uniformly.} \\
				% aka. & \\
				motivation & avoid case distinctions for primitive and compound objects                                                                                                                \\
				idea       & create a common abstract super class enabling a unified access

		\begin{example}{Example -- Formula Tree}
			\begin{itemize}
				\item Components are formula parts
				\item Leaves are literals/numbers
				\item Operators (sum, product) are composites
				\item Formula parts can be arbitrarily nested
				\item Parents can always assume that children provide subresult in compute()

\begin{frame}[fragile]
	\frametitle{\insertsubsection{} -- vs Naive}

	\begin{fancycolumns}[widths={50,50}, t]
		\centering \textbf{Composite Pattern}
		\begin{example}{Class}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\small]
public class Product extends FormulaComponent { ... }

public class Literal extends FormulaComponent { ... }
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\begin{example}{Compute Method}
			\begin{lstlisting}[language=Java, basicstyle=\small]
@Override
public int compute() {
	int product = 1;
	for (FormulaComponent component : getChildren()) {
		product = product * component.compute();
	}
	return product;
}
			\end{lstlisting}
		\end{example}
		\nextcolumn
		\centering \textbf{Naive}
		\begin{example}{Class}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\small]
public class Product { ... }

public class Literal { ... }
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\begin{example}{Compute Method}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\small]
public int compute() {
	int product = 1;
	for (Object element : getChildren()) {
		if (element instanceof Product) {
			product = product * ((Product) element).compute();
		} else if (element instanceof Sum) {
			product = product * ((Sum) element).compute();
		} else if (element instanceof Literal) {
			product = product * ((Literal) element).value;
		}
	}
	return product;
}
		\end{lstlisting}
			\vspace{-3mm}
		\end{example}
	\end{fancycolumns}
\end{frame}

\subsection{Decorator Pattern}
\begin{frame}{\insertsubsection}
	\begin{fancycolumns}
		\begin{definition}{Decorator \mysource{\gof}}
			\setlength\tabcolsep{1mm}
			\begin{tabularx}{\textwidth}{rX}
				intent     & \mycite{Attach additional responsibilities to an object dynamically. Decorators provide a flexible alternative to subclassing for extending functionality.} \\
				aka.       & wrapper (again)                                                                                                                                             \\
				motivation & avoid explosion of static classes when combining all additional behaviors with all applicable classes                                                       \\
				idea       & create decorators and components with the same interface, whereas decorators forward behavior whenever feasible
			\end{tabularx}
		\end{definition}

		\decorator{width=\linewidth}
	\end{fancycolumns}
\end{frame}

\begin{frame}[fragile]
	\frametitle{\insertsubsection -- vs Naive}
	\begin{fancycolumns}[widths={50,50}, t]
		\centering \textbf{Decorator Pattern}
		\begin{example}{Class}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\small]
public CauseLogger(AbstractLogger wrappedLogger) {
	super(wrappedLogger);
	this.wrappedLogger = wrappedLogger;
}
// Instantiate a variant
AbstractLogger motivationalCauseLogger = new MotivationalLogger(new CauseLogger(new BaseLogger()));
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\begin{example}{Creating Messages}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\small]
@Override
public String createMessage(Exception e) {
	return wrappedLogger.createMessage(e) + " " + e.getStackTrace().toString();
}
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\nextcolumn
		\centering \textbf{Naive}
		\begin{example}{Class}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\footnotesize]
public Logger(boolean beMotivational, boolean printCause, boolean printTrace) {
	this.beMotivational = beMotivational;
	this.printCause = printCause;
	this.printTrace = printTrace;
}
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\begin{example}{Creating Messages}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\footnotesize]
public String createMessage(Exception e) {
	String message = e.getMessage();
	if (beMotivational) {
		message = "Come on. You got this! " + message;
	}
	if (printCause) {
		message = message + " " + e.getCause() == null ? "Root cause" : e.getCause().toString();
	}
	if (printTrace) {
		message = message + " " + e.getStackTrace().toString();
	}
	return message;
}
		\end{lstlisting}
			\vspace{-3mm}
		\end{example}
	\end{fancycolumns}
\end{frame}
