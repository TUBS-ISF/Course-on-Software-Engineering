\subsection{Common Optimizations \& Considerations for Software Development}

\begin{frame}[fragile]
	\frametitle{Loop Unrolling}
	\begin{fancycolumns}
		\begin{motivation}{Problem Tackled}
			Overhead of loop control:
			\begin{itemize}
				\item create and initialize loop variable
				\item check loop condition
				\item \emph{run loop body}
				\item increment loop variable
				\item repeat from check loop condition
			\end{itemize}

		\end{motivation}
		\begin{definition}{{Loop Unrolling}}
			\begin{itemize}
				\item Static number of loop runs $\rightarrow$ unroll loop
				\item \emph{Unroll} = repeat loop body and adjust according to loop variable
			\end{itemize}
		\end{definition}
		\nextcolumn
		\begin{example}{Example Loop}
			\begin{lstlisting}[language=Java,basicstyle=\normalsize]
// before loop unrolling
for (int i = 0; i++; i < 3) { 
	a[i] = i; 
}
// after loop unrolling
a[0] = 0; 
a[1] = 1; 
a[2] = 2;
			\end{lstlisting}
		\end{example}
		\begin{note}{Considerations for Developing Software}
			\textbf{smells to avoid}: duplicated code, long method\\\textbf{compiler optimization}: loop unrolling (if number of runs is statically known and small enough)
		\end{note}
	\end{fancycolumns}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Method Inlining}
	\begin{fancycolumns}
		\begin{motivation}{Problem Tackled}
			Overhead of preparing and executing method calls:%
			\begin{itemize}
				\item pass parameters and return address
				\item save registers
				\item restore registers, return to caller
			\end{itemize}
		\end{motivation}
		\begin{note}{Considerations for Developing Software}
			\textbf{smells to avoid}: duplicated code, long method \\
			\textbf{compiler optimization}: method inlining
		\end{note}

			\vspace{-3mm}

// before method inlining
int sum = add(1,2);
int add(int a, int b) {

}
// after method inlining
int sum = 1 + 2;

		\end{example}

\begin{frame}[fragile]
	\frametitle{Arithmetic Optimizations}
	\begin{fancycolumns}
		\begin{motivation}{Problem Tackled}
			Equivalent expressions with different costs:
			\begin{itemize}
				\item Operators with different costs $\rightarrow$ Strength reduction
				\item Operations on constants $\rightarrow$ Constant folding
				\item Common subexpressions $\rightarrow$ Common subexpression elimination
				\item Superfluous computation steps $\rightarrow$ Dead code elimination
			\end{itemize}
		\end{motivation}
		\begin{note}{Considerations for Developing Software}
			\textbf{principles to consider}: KISS \\\textbf{compiler optimizations}: strength reduction, constant folding, common sub-expression elimination, dead code elimination
		\end{note}
		\nextcolumn
		\begin{example}{Example Arithmetic Optimizations}
			\begin{lstlisting}[language=Java,basicstyle=\footnotesize]
// strength reduction
int x = i * 4; // before
int x = i << 2; // after

// constant folding
int x = 3 * 4; // before
int x = 12; // after

// common subexpression elimination
int b = a;
int x = a * 2;
int y = b * 2;
// after
int x = a * 2; // b and y are not needed
			\end{lstlisting}
		\end{example}
	\end{fancycolumns}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Code Motion}
	\begin{fancycolumns}
		\begin{motivation}{Problem Tackled}
			Unnecessary execution of code:
			\begin{itemize}
				\item Invariant code in loops
				\item Code only needed in one branch of a condition
			\end{itemize}
		\end{motivation}
		\begin{definition}{{Code Motion}}
			Move code to prevent unnecessary execution:
			\begin{itemize}
				\item Move invariant code out of loops
				\item Move code into relevant branches
			\end{itemize}
		\end{definition}
		\begin{note}{Considerations for Developing Software}
			\textbf{smells to avoid}: duplicated code, long method\\\textbf{compiler optimization}: loop unrolling (if number of runs is statically known and small enough)
		\end{note}
		\nextcolumn
		\begin{example}{Example Loop}
			\begin{lstlisting}[language=Java,basicstyle=\scriptsize]
// Loop invariant
for (int i = 0; i++; i < 3) {
	int factor = 2;
	a[i] =  factor * i; 
}
// after
int factor = 2;
for (int i = 0; i++; i < 3) {
	a[i] =  factor * i;
}
// Code only needed in one branch
int a = 2;
int b = 4;
if (condition) {
	int c = a * 2;
} else {
	int d = b * 2;
}
// after
if (condition) {
	int a = 2;
	int c = a * 2;
} else {
	int b = 4;
	int d = b * 2;
}

			\end{lstlisting}
		\end{example}
	\end{fancycolumns}


\end{frame}

\begin{frame}[b]{No Garbage Collection?}
	\vspace{-10mm}
	\begin{fancycolumns}[b]
		\begin{motivation}{Automatic vs Manual Memory Management}
			\begin{itemize}
				\item Automatic
				      \begin{itemize}
					      \item simplified programming, programs
					      \item fewer memory leaks
					      \item improved safety, security, reliability
				      \end{itemize}
				\item Manual
				      \begin{itemize}
					      \item more control over memory management
					      \item tailored memory allocation to specific use case $\rightarrow$ potentially better performance
				      \end{itemize}
			\end{itemize}
		\end{motivation}
		\begin{definition}{Manual Memory Management (e.g., C/C++)}
			\begin{itemize}
				\item allocate memory when needed, free when not needed anymore
				\item point to allocated memory and dereference pointer to access memory
				\item can lead to memory leaks and dangling pointers
			\end{itemize}
		\end{definition}
		\nextcolumn
		\begin{definition}{Garbage Collection (e.g., Java)}
			\begin{itemize}
				\item find objects not referenced anymore
				\item free memory space assigned by them
				\item algorithms: reference counting, mark-and-sweep, copying collection
				\item causes random delays (can be improved by e.g., waiting ofr idle time)
			\end{itemize}
		\end{definition}
		\begin{example}{Ownership and Borrowing (e.g., Rust)}
			\begin{itemize}
				\item Each value has a single owner $\rightarrow$ dropped when owner goes out of scope
				\item Ownership can be transferred or temporarily lent (borrowing)
				\item Complex rules to ensure memory safety when borrowing
			\end{itemize}
		\end{example}
	\end{fancycolumns}
\end{frame}
