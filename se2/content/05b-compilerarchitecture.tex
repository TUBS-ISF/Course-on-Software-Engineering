\subsection{Lexing}

\begin{frame}
	\frametitle{\insertsubsection}

	\begin{fancycolumns}[widths={50,50}]
		\begin{definition}{Lexing with Finite Automata (NFA/DFA)}
			\begin{itemize}
				\item Define token classes with regular expressions:
				      \begin{itemize}
					      \small
					      \item int : \texttt{[1..9][0..9]*}
					      \item double : \texttt{[1..9][0..9]*.[0..9]*}
					      \item string: \texttt{"[a-zA-Z][a-zA-Z0..9]*"}
					      \item identifier: \texttt{[a-zA-Z][a-zA-Z0..9]*}
					      \item Keywords
					            \begin{itemize}
						            \small
						            \item public : \texttt{public}
						            \item class : \texttt{class}
					            \end{itemize}
					      \item ...
				      \end{itemize}
				\item Construct NFA with accepting states for each token class
				\item Categorize input into tokens classes
				\item Open questions (for dedicated lecture):
				      \begin{itemize}
					      \item How to construct the automata?
					      \item How to handle errors (e.g., \texttt{1.})?
					      \item How to resolve ambiguities (e.g., is \texttt{int} a keyword or an identifier?)
				      \end{itemize}
			\end{itemize}
		\end{definition}

		\begin{example}{Context-Free Grammar for Simple Program}
			\vspace{-3mm}
			\footnotesize
			\[
				\begin{array}{rcl}
					Program       & \to & StatementList                                              \\
					StatementList & \to & Statement\;StatementList\;\mid\;\epsilon                   \\
					Statement     & \to & Type\;Identifier\;=\;Function                              \\
					Type          & \to & \texttt{int}\;\mid\;\texttt{double}\;\mid\;\texttt{string} \\
					Function      & \to & Identifier\;\texttt{()}                                    \\
					Identifier    & \to & \text{[a-zA-Z][0-9a-zA-Z]*}                                \\
				\end{array}
			\]

	\begin{fancycolumns}[widths={40}]
		\begin{note}{Motivation}
			\begin{itemize}
				\item hard to establish \emph{new programming languages} as compilers needed for many architectures
				\item hard to establish \emph{new architectures} as as compilers needed for many programming languages
