\begin{frame}
    \frametitle{\insertsubsection{}}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Goal}
            \begin{itemize}
                \item Separate data and presentation
            \end{itemize}
        \end{motivation}
        \begin{definition}{Model-View-Controller (MVC)}
            \begin{itemize}
                \item Model: Hold data, provide query operations on it, and notify changes to view
                \item View: Present data to the user, and capture user input
                \item Controller: React to user input $\rightarrow$ manipulate the model and select what to present to the user
            \end{itemize}
        \end{definition}
        \nextcolumn
        \mvcdefault{}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Example}
    \begin{fancycolumns}[widths={45,55}]
        \begin{definition}{MVC for a To-Do List Application}
            \begin{itemize}
                \item Model: Holds list of tasks, and provides operations to add, remove, and query tasks
                \item View: Presents the list of tasks to the user, and captures user input for adding/removing tasks
                \item Controller: Update model according to user input, alter view (e.g., focus on incomplete tasks)
            \end{itemize}
        \end{definition}
        \nextcolumn
        \begin{figure}
            \centering
            \begin{tikzpicture}
                \node (model) at (0,0) [draw, minimum width=2cm, minimum height=1cm] {Todo List};
                \node (view) at (6,0) [draw, minimum width=2cm, minimum height=1cm] {Todo App GUI};
                \node (controller) at (3,-4.5) [draw, minimum width=2cm, minimum height=1cm] {Controller};

                \draw[->] (controller) to node[above, midway, sloped] {Manipulates} (model);

                \draw[->, bend left=12] (controller) to node[ above, midway, sloped] {Select Presentation} (view);
                \draw[->, bend left=12] (view) to node[below, midway, sloped] {(User) Input} (controller);

                \draw[->, bend left=12] (model) to node[above, midway, sloped] {Notify change} (view);
            \end{tikzpicture}
        \end{figure}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Considerations}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Advantages}
            \begin{itemize}
                \item Separation of concerns $\rightarrow$ easier to maintain and extend
                \item Parallel development of model, view, and controller
                \item Reusability of components (e.g., different views for the same model)
            \end{itemize}
        \end{motivation}
        \begin{note}{Disadvantages}
            \begin{itemize}
                \item Initial design effort
                \item Overhead especially for simple applications
            \end{itemize}
        \end{note}

        \nextcolumn
        \mvcdefault{}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{}}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Goal}
            \begin{itemize}
                \item Encapsulate data management in a central repository
                \item Communication over shared data
            \end{itemize}
        \end{motivation}
        \begin{definition}{Repository Architecture}
            \begin{itemize}
                \item Repository = central data storage
                \item Unified interface for data management
                \item Components communicate by reading/writing to repository
            \end{itemize}
        \end{definition}
        \nextcolumn
        \repositorydefault{}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Example}
    \begin{fancycolumns}[widths={45,55}]
        \begin{definition}{Repository Architecture for a University Information System}
            \begin{itemize}
                \item Repository holds all student data (e.g., personal information, course enrollments, grades)
                \item Components all interact with the repository to read/write student data (e.g., register grades or enroll in courses)
                \item Communication between components only via the means of data stored in the repository (e.g, examiner updates grade $\rightarrow$ student portal can read updated grade from repository)
            \end{itemize}
        \end{definition}
        \nextcolumn
        \begin{figure}

                \draw[<->] (c1.south) -- node[right, midway] {Read/Write} (c1 |- repo.north);
                \draw[<->] (c2.south) -- node[right, midway] {Read/Write} (c2 |- repo.north);
                \draw[<->] (c3.north) -- node[right, midway] {Read/Write} (c3 |- repo.south);
                \draw[<->] (c4.north) -- node[right, midway] {Read/Write} (c4 |- repo.south);
            \end{tikzpicture}
        \end{figure}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Considerations}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Advantages}
            \begin{itemize}
                \item Separate business logic from data management
                \item Components are independent of each other
                \item Data is centralized $\rightarrow$ easier to persist and share
            \end{itemize}
        \end{motivation}
        \begin{note}{Disadvantages}
            \begin{itemize}
                \item Potentially inefficient way of interaction between components
                \item Repository is single point of failure
                \item Realization in distributed systems can be complex (e.g., how to ensure correct synchronization?)
            \end{itemize}
        \end{note}
        \nextcolumn
        \repositorydefault{}
    \end{fancycolumns}
\end{frame}



\subsection{Pipes and Filters} % Example for organizing data processing

\newcommand{\pipesfiltersdefault}{
    \begin{figure}
        \centering
        \begin{tikzpicture}
            \node (f1) at (0,0) [draw, minimum width=2cm, minimum height=1cm] {Filter 1};
            \node (f2) at (4,0) [draw, minimum width=2cm, minimum height=1cm] {Filter 2};
            \node (f3) at (4,-3) [draw, minimum width=2cm, minimum height=1cm] {Filter 3};

            \draw[->] (f1) to node[above, midway, sloped] {Pipe} (f2);
            \draw[->] (f2) to node[above, midway, sloped] {Pipe} (f3);
        \end{tikzpicture}
    \end{figure}
}

\begin{frame}
    \frametitle{\insertsubsection{}}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Goal}
            \begin{itemize}
                \item Reusability of transformation steps
            \end{itemize}
        \end{motivation}
        \begin{definition}{Pipes and Filters}
            \begin{itemize}
                \item Filter = transformation step
                \item Pipe = data flow between filters
                \item Filters are independent of each other
                \item Filters rely on specific input and output data formats
            \end{itemize}
        \end{definition}
        \nextcolumn
        \pipesfiltersdefault{}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Example}
    \begin{fancycolumns}[widths={45,55}]
        \begin{definition}{Abstract Compiler as Pipes and Filters}
            \begin{itemize}
                \item Filter 1: Parser $\rightarrow$ transforms source code into an abstract syntax tree (AST)
                \item Filter 2: Code Generator $\rightarrow$ transforms AST intomachine language
                \item Filter 3: Compiler optimization on machine language
            \end{itemize}
        \end{definition}
        \nextcolumn
        \begin{figure}
            \centering
            \begin{tikzpicture}
                \node (f1) at (0,0) [draw, minimum width=2cm, minimum height=1cm] {Parser};
                \node (f2) at (4,0) [draw, minimum width=2cm, minimum height=1cm] {Code Generator};
                \node (f3) at (4,-3.5) [draw, minimum width=2cm, minimum height=1cm] {Optimizer};

                \draw[->] (-3,0) to node[above, midway, sloped] {\small Source Code} (f1);
                \draw[->] (f1) to node[above, midway, sloped] {AST} (f2);
                \draw[->] (f2) to node[above, midway, sloped] {\small Machine Language} (f3);
                \draw[->] (f3) to node[above, midway, sloped] {\small Machine Language} (0.5, -3.5);
            \end{tikzpicture}
        \end{figure}
    \end{fancycolumns}
\end{frame}

\begin{frame}
    \frametitle{\insertsubsection{} -- Considerations}
    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Advantages}
            \begin{itemize}
                \item Isolated testing of filters
                \item Simple Reusability, exchangeability and extension of filters
            \end{itemize}
        \end{motivation}
        \begin{note}{Disadvantages}
            \begin{itemize}
                \item Potential overhead caused by restrictions on input/output of filters
            \end{itemize}
        \end{note}
        \nextcolumn
        \pipesfiltersdefault{}
    \end{fancycolumns}
\end{frame}

\begin{frame}

    \begin{fancycolumns}[widths={45,55}]
        \begin{motivation}{Goal}
            \begin{itemize}
                \item Efficient communication between components
            \end{itemize}
        \end{motivation}
        \begin{definition}{Event-Driven Architecture (EDA)}
            \begin{itemize}
                \item \emph{Event} = significant change in state
                \item \emph{Producers} generate events
                \item \emph{Consumers} react to events
                \item \emph{Broker} manages event distribution
                \item There may be components that are both producers and consumers
            \end{itemize}
        \end{definition}

        \begin{motivation}{Advantages}
            \begin{itemize}
                \item Real-time response to events
                \item Loose coupling between components
                \item Adding new components does not affect existing ones
                      \begin{itemize}
                          \item Consumers can subscribe to already produced events
                          \item Producing new events does not affect existing consumers
                      \end{itemize}
            \end{itemize}
        \end{motivation}
        \begin{note}{Disadvantages}
            \begin{itemize}
                \item Complexity of event management
                \item Asynchronous communication introduces challenges (e.g., race conditions)
                \item Events may not be suitable for all types of interactions between components
            \end{itemize}
        \end{note}
