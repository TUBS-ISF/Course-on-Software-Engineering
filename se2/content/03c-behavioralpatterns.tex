\subsection{Observer Pattern}
\begin{frame}{\insertsubsection} % TODO copied from SE1. reuse?
	\begin{fancycolumns}
		\begin{definition}{Observer \mysource{\gof}}
			\setlength\tabcolsep{1mm}
			\begin{tabularx}{\textwidth}{rX}
				intent     & \mycite{Define a one-to-many dependency between objects so that when one object changes state, all its dependents are notified and updated automatically.} \\
				aka.       & dependents, publish-subscribe                                                                                                                              \\
				motivation & avoid coupling of classes (i.e., explicit references such as imports)                                                                                      \\
				example    & no coupling between model and view in model-view-controller architectures                                                                                  \\
				idea       & data object (model) has a list of observers (views) that are notified about changes
			\end{tabularx}
		\end{definition}
		\nextcolumn
		\observer{width=\linewidth}
	\end{fancycolumns}
\end{frame}

\begin{frame}{\insertsubsection{} -- Example}
	\begin{fancycolumns}[widths={35,65}]
		\begin{example}{Example -- Different Modules in a Running Tracker}
			\begin{itemize}
				\item Running tracker with various different features (e.g., social media, long time process)
				\item Many modules that need to be updated when a run is finished
				\item Adding new modules should not require changes to the existing modules
				\item Make main run list observable and let other modules observe it
			\end{itemize}
		\end{example}

		\centering \textbf{Observer Pattern}
		\begin{example}{Observable}
			\vspace{-3mm}
			\makeatletter
			\lst@AddToHook{OnEmptyLine}{\vspace{-0.7\baselineskip}}
			\makeatother
			\begin{lstlisting}[language=Java, basicstyle=\scriptsize]
public class RunTracker {
    List<IRunObserver> observers;
    RunlistState state;

    public void addObserver(IRunObserver observerToAdd) {
        this.observers.add(observerToAdd);
    }

    public void removeObserver(IRunObserver observerToRemove) {
        this.observers.remove(observerToRemove);
    }

    public void notifyObservers() {
        for (IRunObserver observer : observers) {
            observer.update(state);
        }
    }

    public void addRun(Run run) {
        state.runList.add(run);
        notifyObservers();
    }

}
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}
		\begin{example}{Initialize Observers}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\footnotesize]
RunTracker tracker = new RunTracker();
SocialNotifier socialNotifier = new SocialNotifier();
tracker.addObserver(socialNotifier);
			\end{lstlisting}
			\vspace{-3mm}
		\end{example}

\subsection{Visitor Pattern}
\begin{frame}{\insertsubsection}
	\begin{fancycolumns}
		\begin{definition}{Visitor \mysource{\gof}}
			\setlength\tabcolsep{1mm}
			\begin{tabularx}{\textwidth}{rX}
				intent     & \mycite{Represent an operation to be performed on the elements of an object structure.}                                                \\
				%aka. & \\
				motivation & avoid that the implementation of an operation is spread over several classes                                                           \\
				example    & compiler operates on code in a tree structure, but has numerous operations (e.g., code generation, checking for names and type errors) \\
				idea       & each data object has a generic method to pass a visitor object and each operation is implemented in a dedicated visitor
			\end{tabularx}
		\end{definition}
		\nextcolumn
		\visitor{width=\linewidth}
	\end{fancycolumns}
\end{frame}

\begin{frame}{\insertsubsection{} -- Example}
	\begin{fancycolumns}[widths={35,65}]
		\begin{example}{Example -- Visiting a Formula Tree}
			\begin{itemize}
				\item Various operations could be required on a formula (e.g., compute, print)
				\item External traversal requires case many case distinctions
				\item Embedding operations in formula components violates separation of concerns
				\item Use visitor pattern to separate operations from formula elements
			\end{itemize}
		\end{example}

		\centering \textbf{Naive}
		\begin{example}{Operations attached to Objects}
			\vspace{-3mm}
			\begin{lstlisting}[language=Java, basicstyle=\scriptsize]
public abstract class FormulaComponent {
    public abstract int compute();
    public abstract String print();
    public abstract FormulaComponent[] getChildren();
}

public class Sum extends FormulaComponent {
	// ...
    @Override
    public int compute() {
        int sum = 0;
        for (FormulaComponent element : getChildren()) {
            sum += element.compute();
        }
        return sum;
    }

    @Override
    public String print() {
        String print = "+(";
        for (FormulaComponent child : getChildren()) {
            print += (String) child.print() + ",";
        }
        print = print.substring(0, print.length() - 1) + ")";
        return print;
    }   
}
		\end{lstlisting}
